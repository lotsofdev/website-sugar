# isModuleSystem
<div class="_metas">
<div class="_type"><span class="_type-label">Type:</span><span class="_type-value">Function</span></div>
<div class="_status"><span class="_status-label">Status:</span><span class="_status-value -stable">stable</span></div>

<div class="_description">

</div>
<div class="_params">
## Params
<ol class="_list">
<li class="_item">
<span class="_name">systemNames<span class="_required">*</span></span><span class="_default"></span> <span class="_type">&lpar;&apos;esm&apos;&vert;&apos;cjs&apos;&rpar;&lsqb;&rsqb;</span>
<p class="_description">An array of system names to check against</p>
</li>
</ol>
</div>
<div class="_return">
## Return
<ol class="_list">
<li class="_item">
<span class="_description">true if the current system is one of the passed system names, false otherwise</span><span class="_default"></span><span class="_type">Boolean</span>
</li>
</ol>
</div>
<div class="_examples">
## Example
```js
import { __isModuleSystem } from '@lotsof/sugar/is';
__isModuleSystem('esm'); // => true
__isModuleSystem('cjs'); // => false
```
