<div class="_examples">
## Example
```js
import { __propertyProxy } from '@lotsof/sugar/object';
const myObject = {
		title : 'World'
};
// create the proxy
__propertyProxy(myObject, 'title', {
		get : (value) => {
			return `Hello ${value}`;
		},
		set : (value) => {
			return `Youhou ${value}`;
		}
});
console.log(myObject.title) => 'Hello World';
myObject.title = 'Universe';
console.log(myObject.title) => 'Hello Youhou Universe';
```
